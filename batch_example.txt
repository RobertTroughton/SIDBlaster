# SIDBlaster Batch Processing Example
# This example demonstrates several batch operations:
# 1. Convert a SID file to PRG with player
# 2. Create a trace log of the original SID
# 3. Relocate the SID to $2000
# 4. Create a trace log of the relocated SID
# 5. Verify that both trace logs match

[Task1]
type=Trace
input=D:\SID\*.sid
output=temp\original-trace.bin
format=binary
frames=30000

[Task2]
type=Relocate
input=SID\*.sid
output=temp\music-rel0800.sid
address=0x0800

[Task3]
type=Trace
input=temp\music-rel0800.sid
output=temp\relocated-trace.bin
format=binary
frames=30000

[Task4]
type=Verify
original=temp\original-trace.bin
relocated=temp\relocated-trace.bin
reportFile=verification.log

[Task5]
type=Convert
input=temp\music-rel0800.sid
output=D:\SID-Relocated\*.prg
format=PRG
player=SimpleRaster
playerAddr=0x0400
